git add <path> - путь к файлу, добавить файл в  локальный репозиторий
git add . - добавляются все файлы в текущей папке
git add ./название файла - добавление файла 
git add ../название файла - добавление файла из папки на уровень выше
git commit -m 'имя комита' - финальное лобавление изменений в локальный репозиторий 
git push - добавление изменений в удаленный репозиторий 
git pull --rebase - добавление твоих изменений после синхронизации с удаленным репозиторием /
